3- select CONCAT(u.first_name," ",u.last_name) as "User Name", COUNT(f.id) as "Number of Friends"
from users u, users f  , friendships  fs  
where  u.id=fs.user_id && f.id=fs.friend_id
ORDER BY "Number of Friends" desc limit 1
